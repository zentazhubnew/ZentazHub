repeat task.wait() until game:IsLoaded()

_G.ZentazHub = _G.ZentazHub or {}

_G.ZentazHub.Script = {
    url = "https://raw.githubusercontent.com/accsultan428-sys/8zisisisis/refs/heads/main/sjsjejeje.lua."
}

function _G.ZentazHub.Load(url)
    local success, result = pcall(function()
        return game:HttpGet(url)
    end)
    if success and result then
        local func, err = loadstring(result)
        if func then
            pcall(func)
            print("✅ ZentazHub Loaded!")
        else
            warn("❌ Compile error: " .. tostring(err))
        end
    else
        warn("❌ Failed to load: " .. tostring(result))
    end
end

-- Load script
_G.ZentazHub.Load(_G.ZentazHub.Script.url)
